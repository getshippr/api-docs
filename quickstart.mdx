## Add Shippr to your project
Works both on the browser and node.js
<CodeGroup>
```js YARN
yarn add @shippr/client
```
```js NPM
npm install @shippr/client
```
</CodeGroup>

### Start building

<AccordionGroup>
  <Accordion icon="rectangle-terminal" title="Pub">
    ```js index.ts
import shippr from "@shippr/client"

const client = shippr("APPID", "APIKEY")
client.publish("my-channel", { data: "something happened" })
      
    ```
  </Accordion>
  <Accordion icon="rectangle-terminal" title="Sub">
      ```js index.ts

    import shippr from "@shippr/client"

    const client = shippr("APPID", "APIKEY")
    const myWatcher = client.subscribe("my-channel")
    myWatcher.on((data, err) => {
        doSomethingFun()
    })

          ```

  </Accordion>
  <Accordion icon="rectangle-terminal" title="Webhooks">
  <CodeGroup>
      ```sh call our webhook
      curl -X POST -H "Content-Type: application/json" -d '{"bar": "foo"}' "https://push.getshippr.com/hooks/my-channel-test?apiKey=YOUR_KEY&appId=YOUR_APP"
      ```
      ```js receiver.ts

    const myWebhook = client.subscribe("my-channel-test")
    myWebhook.on((data, err) => {
        doSomethingFun()
    })

          ```
    </CodeGroup>
  </Accordion>
</AccordionGroup>
