## ShipprProvider

To avoid having to have to add credentials everywhere, you need to wrapp your application with a ShipprProvider.

For instance, if you are using Next, we recommend adding it in `_app.tsx`:

```jsx
import { ShipprProvider } from '@shippr/react'

export default function App({ Component, pageProps: { ...pageProps } }) {
  return (
    <ShipprProvider appId={"your_app_id"} apiKey={"your_api_key"} options={{...options}}>
        <Component {...pageProps} />
    </ShipprProvider>
  )
}
```


## useSharedState Hook

Shippr provides a React hook hides the complexity of connecting to a channel and maintaining the value

<CodeGroup>

```jsx with useState
import React, { useState } from "react"
import shippr from "@shippr/react"
const client = shippr("APPID", "APIKEY")

export default function Demo() {
  const [value, setValue] = useState(0)
  
  useEffect(() => {
    const fetch = async () => {
      const myWatcher = await client.subscribe("my-update")
      myWatcher.on((data, err) => {
          setValue(data)
      })
    }
    fetch()
  }, [])
  
  return (
    <div>
      {value}
    </div>
  )
}

```
```jsx with useSharedState
import React from "react"
import reactClient from "@shippr/react"
const { useSharedState } = reactClient("APPID", "APIKEY")

export default function Demo() {
  const [value, setValue] = useSharedState(0, "my-shared-state-channel")
  return (
    <div>
      {value}
    </div>
  )
}

```
</CodeGroup>
